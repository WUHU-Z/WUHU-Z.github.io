3.1.1单源最短路的建图方式

主要是在问题中将单源最短路模型抽象出来

1.Acwing 1129.热浪

题目：

思路：

单源最短路模型：

1.朴素迪杰斯特拉算法

2.堆优化的杰斯塔拉算法

3.spfa算法

代码（spfa）：

#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=2510,M=6200*2+10;//N表示城镇数，M表示路径数
int n,m,S,T;
int h[N],e[M],w[M],ne[N],idx=0;
int dist[N];//距离
int q[N];//循环队列
bool st[N];//判重数组

void add(int a,int b,int c)
{
    e[idx]=b,w[idx]=c,ne[idx]=h[a],h[a]=idx++;
}

void spfa()
{
    memset(dist,0x3f,sizeof dist);
    dist[S]=0;//初始化起点距离
    int hh=0,tt=1;//初始化循环队列队头队尾
    q[0]=S,st[S]=true;//入队并标记该点已经走过
    while(hh!=tt)
    {
        int t = q[hh++];
        if(hh==N)hh=0;
        st[t]=false;
        for(int i=h[t];~i;i=ne[i])
        {
            int j=e[i];
            if(dist[j]>dist[t]+w[i])
            {
                dist[j]=dist[t]+w[i];
                if(!st[j])
                {
                    q[tt++]=j;
                    if(tt==N)tt=0;
                    st[j]=true;
                }
            }
        }
    }
}

int main()
{
    cin>>n>>m>>S>>T;
    memset(h,-1,sizeof h);
    for(int i=0;i<m;i++)
    {
        int a,b,c;
        cin>>a>>b>>c;
        add(a,b,c);//因为路径是双向的
        add(b,a,c);
    }
    spfa();
    cout<<dist[T]<<endl;
    return 0;
}

代码（朴素DIjkstra）：

#include<iostream>
#include<cstring>
#include<algorithm>
#define INF 9999999
using namespace std;
const int N=2510;//N表示城镇数
int e[N][N];
int dist[N];
int book[N];
int n,m,t1,t2,t3,S,T;
int main()
{

    cin>>n>>m>>S>>T;

    //初始化图
    for(int i=1;i<=n;i++)
    for(int j=1;j<=n;j++)
        if(i==j)e[i][j]=0;
        else e[i][j]=INF;

    //读入边
    for(int i=0;i<m;i++)
    {
        cin>>t1>>t2>>t3;
        e[t1][t2]=t3;
        e[t2][t1]=t3;
    }
    //初始化dis数组
    for(int i=1;i<=n;i++)
        dist[i]=e[S][i];
    //book数组初始化
    memset(book,0,sizeof book);
    book[S]=1;
    for(int i=1;i<=n-1;i++)
    {
        //找到离更新点最近的顶点，默认初始的更新点为1
        int res=INF;
        int u;
        for(int j=1;j<=n;j++)
        {
            if(book[j]!=1&&dist[j]<res)
            {
                res=dist[j];
                u=j;
            }
        }
        //表示新的更新点已经找出
        book[u]=1;

        //以新的更新点为基准开始更新
        for(int v=1;v<=n;v++)
        {
            if(e[u][v]<INF)
            {
                if(dist[v]>dist[u]+e[u][v])
                    dist[v]=dist[u]+e[u][v];
            }
        }
    }
    cout<<dist[T]<<endl;
    return 0;
}

代码（堆优化Dijkstra）：

/*
这个堆优化其实很容易理解，将枚举最小值改为从堆中取出最小值，改变dis时入堆即可

用单调队列维护时必须有两个值：点的编号和当前的距离

以距离为标准从小到大排序， 每次去除最小的
*/

#include<iostream>
#include<algorithm>
#include<cstring>
#include<queue>
using namespace std;
const int N=3e3,M=2e4;
typedef pair<int,int> PII;
int h[N],w[M],e[M],ne[M],idx;//设立邻接表
bool book[N];
priority_queue<PII,vector<PII>,greater<PII>>heap;

void add(int a,int b,int c)//加边
{
    e[idx]=b,ne[idx]=h[a],w[idx]=c,h[a]=idx++;
}

int dijkstra(int st,int ed)
{
    int dist[M];//初始化距离数组
    memset(dist,0x3f,sizeof dist);
    dist[st]=0;
    book[st]=true;//标记该点走过
    heap.push({0,st});
    while(heap.size())
    {
        auto t=heap.top();
        heap.pop();
        book[t.second]=false;
        for(int i=h[t.second];~i;i=ne[i])
        {
            int j=e[i];
            if(dist[j]>t.first+w[i])
            {
                dist[j]=t.first+w[i];
                heap.push({dist[j],j});
            }
        }
    }
    return dist[ed];
}

int main()
{
    int n,m,st,ed;
    cin>>n>>m>>st>>ed;
    memset(h,-1,sizeof h);
    for(int i=0;i<m;i++)
    {
        int a,b,w;
        cin>>a>>b>>w;
        add(a,b,w);
        add(b,a,w);
    }
    int t=dijkstra(st,ed);
    cout<<t<<endl;
    return 0;
}

3.Acwing1127.香甜的黄油（多源汇最短路问题）

题目：

思路：

因为使用Floyd算法会超时，所以放弃

对每一个点使用堆优化Dijkstra，然后dist相加比较找到最小点。

spfa会被卡空间…

代码（spfa死了，我选择堆优化）：

#include<iostream>
#include<cstring>
#include<queue>
#define INF 0x3f3f3f3f

using namespace std;
typedef pair<int,int> PII;
const int N=510;
const int P=810;
const int C=1450*2;
int n,p,c,t1,t2,t3;
int z[N];
int h[P],e[C],ne[C],w[C],idx;
priority_queue<PII,vector<PII>,greater<PII>>heap;
int dist[P];
bool book[P];
void add(int a,int b,int c)
{
    e[idx]=b,w[idx]=c,ne[idx]=h[a],h[a]=idx++;
}

int dijkstra(int start)
{
    memset(dist,0x3f,sizeof dist);
    memset(book,0,sizeof book);
    dist[start]=0;
    heap.push({0,start});
    while(heap.size())
    {
        auto ovo=heap.top();
        heap.pop();
        if(book[ovo.second])continue;
        book[ovo.second]=1;
        for(int i=h[ovo.second];~i;i=ne[i])
        {
            int j=e[i];
            if(dist[j]>ovo.first+w[i])
            {
                dist[j]=ovo.first+w[i];
                heap.push({dist[j],j});
            }
        }
    }
    int res=0;
    for(int i=1;i<=n;i++)
    {
        int j=z[i];
        if(dist[j]==INF)return INF;
        else res+=dist[j];
    }
    return res;
}

int main()
{
    memset(h,-1,sizeof h);
    cin>>n>>p>>c;
    for(int i=1;i<=n;i++)cin>>z[i];
    for(int i=1;i<=c;i++)
    {
        cin>>t1>>t2>>t3;
        add(t1,t2,t3);
        add(t2,t1,t3);
    }
    int minn=INF;
    for(int i=1;i<=p;i++)
    {
        int res=dijkstra(i);
        minn=min(res,minn);
    }
    cout<<minn<<endl;
    return 0;
}

4.Acwing1126最小花费

题目：

思路：

以上便于理解题目，实际还是求单源最长路即可

注意：

  1. 因为此处数据过大（10^5），所以使用scanf输入
  2. 因为精确到小数点后8位，所以使用printf输出
  3. 对于乘积类问题，可以转化为单源最大路问题 =）

总结规律：

加法最小值

  1. 无负权边：dijkstra 理由：算法的前提是st数组里的都是已经确定是最短路的，若存在负权边会破坏这个前提。
  2. 有负权边：spfa

加法最大值

  1. 自我感觉spfa能求，还未验证过

乘法最小值

  1. >=1（相当于无负权边）：dijkstra 理由：算法的前提是st数组里的都是已经确定最小值的，若存在<1的话会破坏这个前提。
  2. >0（相当于有负权边）：spfa

乘法最大值

  1. 0-1：dijkstra
  2. >0:spfa

代码（spfa）：

#include<iostream>
#include<cstring>
using namespace std;
const int N=2010;
const int M=2e5+5;
int n,m,S,T;
double dist[N];
bool st[N];
int e[M],ne[M],h[N],idx;
double w[M];
int q[N];
void add(int a,int b,int c)
{
    e[idx]=b,ne[idx]=h[a],w[idx]=(100.0-c)/100,h[a]=idx++;
}

void spfa()
{
    dist[S]=1;
    st[S]=1;
    q[0]=S;
    int hh=0,tt=1;
    while(hh!=tt)
    {
        int z=q[hh++];
        if(hh==N)hh=0;
        st[z]=0;
        for(int i=h[z];~i;i=ne[i])
        {
            int j=e[i];
            if(dist[j]<dist[z]*w[i])
            {

                dist[j]=dist[z]*w[i];
                if(!st[j])
                {
                    q[tt++]=j;
                    if(tt==N)tt=0;
                    st[j]=1;
                }
            }
        }
    }
}

int main()
{
    scanf("%d%d",&n,&m);
    memset(h,-1,sizeof h);
    while(m--)
    {
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
        add(b,a,c);
    }
    scanf("%d%d",&S,&T);
    spfa();
    printf("%.8lf\n",100/dist[T]);
    return 0;
}

5.Acwing920最优乘车

题目：

思路：

可以将题目看做是乘车的最少次数-1

将每一个站牌看做一个点，对于每一条公交路线，将线上每一个点，与它能到的点连接上一条权重为1的边，表示从该点到那个点只需乘车一次即可

然后单源最短路（没想到我居然要用BFS）即可，需要特判1=N

注意：

因为没有说明每一条线路要经过多少站，所以这里需要使用sstream来读入相关信息

代码（很像spfa的bfs）：

#include<iostream>
#include<cstring>
#include<sstream>

using namespace std;
const int N=510;
int m,n;
bool g[N][N];
int dist[N];
int stop[N];
int q[N];
bool st[N];//可以没有

void bfs()
{
    int hh=0,tt=1;
    memset(dist,0x3f,sizeof dist);
    q[0]=1;
    dist[1]=0;
    st[1]=1;
    while(hh<=tt)
    {
        int t=q[hh++];
        st[t]=0;
        for(int i=1;i<=n;i++)
            if(g[t][i]&&dist[i]>dist[t]+1)
        {
            dist[i]=dist[t]+1;
            if(!st[i])
            {
                q[++tt]=i;
                st[i]=1;
            }
        }
    }
}

int main()
{
    cin>>m>>n;
    string line;
    getline(cin,line);
    while(m--)
    {
        getline(cin,line);
        stringstream ssin(line);
        int cnt=0,p;
        while(ssin>>p)stop[cnt++]=p;
        for(int i=0;i<cnt;i++)
            for(int j=i+1;j<cnt;j++)
            g[stop[i]][stop[j]]=true;
    }

    bfs();
    if(dist[n]==0x3f3f3f3f)cout<<"NO"<<endl;
    else cout<<max(dist[n]-1,0)<<endl;
    return 0;
}

6.Acwing903.昂贵的聘礼

题目：

年轻的探险家来到了一个印第安部落里。

在那里他和酋长的女儿相爱了，于是便向酋长去求亲。

酋长要他用 10000 个金币作为聘礼才答应把女儿嫁给他。

探险家拿不出这么多金币，便请求酋长降低要求。

酋长说：”嗯，如果你能够替我弄到大祭司的皮袄，我可以只要 8000 金币。如果你能够弄来他的水晶球，那么只要 5000 金币就行了。”

探险家就跑到大祭司那里，向他要求皮袄或水晶球，大祭司要他用金币来换，或者替他弄来其他的东西，他可以降低价格。

探险家于是又跑到其他地方，其他人也提出了类似的要求，或者直接用金币换，或者找到其他东西就可以降低价格。

不过探险家没必要用多样东西去换一样东西，因为不会得到更低的价格。

探险家现在很需要你的帮忙，让他用最少的金币娶到自己的心上人。

另外他要告诉你的是，在这个部落里，等级观念十分森严。

地位差距超过一定限制的两个人之间不会进行任何形式的直接接触，包括交易。

他是一个外来人，所以可以不受这些限制。

但是如果他和某个地位较低的人进行了交易，地位较高的的人不会再和他交易，他们认为这样等于是间接接触，反过来也一样。

因此你需要在考虑所有的情况以后给他提供一个最好的方案。

为了方便起见，我们把所有的物品从 1 开始进行编号，酋长的允诺也看作一个物品，并且编号总是 1。

每个物品都有对应的价格 P，主人的地位等级 L，以及一系列的替代品Ti和该替代品所对应的”优惠” Vi。

如果两人地位等级差距超过了 M，就不能”间接交易”。

你必须根据这些数据来计算出探险家最少需要多少金币才能娶到酋长的女儿。

输入格式

输入第一行是两个整数 M，N，依次表示地位等级差距限制和物品的总数。

接下来按照编号从小到大依次给出了 N 个物品的描述。

每个物品的描述开头是三个非负整数 P、L、X，依次表示该物品的价格、主人的地位等级和替代品总数。

接下来 X 行每行包括两个整数 T 和 V，分别表示替代品的编号和”优惠价格”。

思路：

代码：

#include<iostream>
#include<cstring>

using namespace std;
const int N=110;
const int INF=0x3f3f3f3f;

int n,m;
int d[N][N],level[N];
int dist[N];
bool st[N];

inline int dijkstra(int down,int up)//在对应区间中，从花费最小的开始枚举
{
    memset(dist,0x3f,sizeof dist);
    memset(st,0,sizeof st);
    dist[0]=0;
    for(int i=1;i<=n;i++)//因为最后一个花费一定是酋长女儿的10000，所以不用更新n+1
    {
        int t= -1;
        for(int j=0;j<=n;j++)
        {
                if(!st[j]&&(t==-1||dist[t]>dist[j]))
                t=j;
        }
        st[t]=true;
        for(int j=1;j<=n;j++)
        {
            if(level[j]>=down&&level[j]<=up)//只有该点于等级范围内时才开始更新
            {
                dist[j]=min(dist[j],dist[t]+d[t][j]);
            }
        }
    }
    return dist[1];
}

int main()
{
    cin>>m>>n;
    memset(d,0x3f,sizeof d);
    for(int i=1;i<=n;i++)
    {
        int price,cnt;
        cin>>d[0][i]>>level[i]>>cnt;
        while(cnt--)
        {
            int id,cost;
            cin>>id>>cost;
            d[id][i]=cost;
        }
    }
    int res=INF;
    for(int i=level[1]-m;i<=level[1];i++)//枚举所有可行区间的最小值
    {
        res=min(res,dijkstra(i,i+m));
    }
    cout<<res<<endl;
    return 0;
}
