A - Sudoku

题目：

POJ - 3076

思路：

我的思路：

利用二进制存图，然后通过lowbit和&找到对应数字并更新，由于没有剪枝爆了

大佬的思路：

①肯定是选择可填入的字母最少的位置开始dfs，这样分支比较少

②如果一个位置只剩一个字母可以填，就填上这个字母（废话）

③如果所有的字母不能填在该行（列、十六宫格），立刻回溯

④如果某个字母只能填在该行（列、十六宫格）的某处，立刻填写

首先除了dfs的结束条件外，一次写下对位置的②剪枝，对行，列，十六宫格的③、④剪枝，因为②、④剪枝填了一个字母，需要再次判断结束条件，

然后就是对可能性最小的位置进行dfs

#include<iostream>
#include<cstring>

using namespace std;

int maps[17][17];//存图
int table[17][17];//记录每一个位置不可以放哪些数字

int num;//记录存在的字符数量

void put_in(int x,int y,int k)
{
    num++;
    maps[x][y] = k;
    //对应点上更新
    table[x][y] |= 1<<k;
    //对应行列上更新
    for(int i=1; i<=16; i++)
    {
        table[i][y] |= 1<<k;
        table[x][i] |= 1<<k;
    }
    //对应矩阵中更新
    //防止4,8等情况影响更新
    int r = (x-1)/4*4+1;
    int c = (y-1)/4*4+1;
    for(int i=0; i<4; i++)
    {
        for(int j=0; j<4; j++)
        {
            table[r+i][c+j] |= 1<<k;
        }
    }
}

void print()//输出，代表终结
{
    for(int i=1; i<=16; i++)
        {
            for(int j=1; j<=16; j++)
            {
                printf("%c",maps[i][j]+'A');
            }
            puts("");
        }
        puts("");
}

//判断该点是否有唯一数字
//如果有，则返回对应数字；反之，则返回-1
int _count(int x)
{
    for(int i=0; x; i++)//查找
    {
        if(x & 1)
        {
            if(x>>1 == 0)
                return i;
            return -1;
        }
        x >>= 1;
    }
    return -1;//理论上不可能运行到这一句
}


int row(int x,int k)//尝试用数字k更新x行
{
    int t = -1;
    for(int y=1; y<=16; y++)
    {
        if(maps[x][y] == k)return -1;//如果被使用过返回-1
        if(maps[x][y] >= 0)continue;//如果该点有确定数字，continue
        if((table[x][y]&1<<k) == 0)//如果该点可填入该数字
        {
            if(t != -1)return -1;//如果有多个列都满足k的要求，返回-1
            t = y;
        }
    }
    if(t != -1)return t;//如果找到唯一列满足k的要求
    return -2;//如果没有找到满足填入k要求的列
}

int col(int y,int k)//尝试用数字k更新y列
{
    int t = -1;
    for(int x=1; x<=16; x++)
    {
        if(maps[x][y] == k)return -1;//如果被使用过返回-1
        if(maps[x][y] >= 0)continue;//如果该点有确定数字，continue
        if((table[x][y]&1<<k) == 0)//如果该点可填入该数字
        {
            if(t != -1)return -1;//如果有多个行都满足k的要求，返回-1
            t = x;
        }
    }
    if(t != -1)return t;//如果找到唯一行满足k的要求
    return -2;  //如果没有找到满足填入k要求的行
}

void cuble(int r,int c,int k,int &x,int &y)//尝试用数字k更新该矩阵
{
    x = -2;
    for(int i=0; i<4; i++)
    {
        for(int j=0; j<4; j++)
        {
            if(maps[i+r][j+c] == k)
            {
                x=-1;
                return;
            }
            if(maps[i+r][j+c] >= 0)
                continue;
            if((table[i+r][j+c]&1<<k) == 0)
            {
                if(x != -2)
                {
                    x=-1;
                    return;
                }
                x = i;
                y = j;
            }
        }
    }
}

int cal(int x)//查找有多少个数字无法使用
{
    int cnt = 0;
    while(x)
    {
        if(x&1)
            cnt++;
        x >>= 1;
    }
    return cnt;
}

bool dfs()
{
    if(num==256)//如果所有点都已填上字符
    {
        print();
        return true;
    }
    for(int i=1; i<=16; i++)//遍历所有没有确定数字的点
    {
        for(int j=1; j<=16; j++)
        {
            if(maps[i][j] >= 0)continue;
            int k = _count(table[i][j]);//查看是否能找出
            if(k != -1)put_in(i,j,k);//如果该点有唯一数字，则填入
        }
    }
    //遍历所有行
    for(int x=1; x<=16; x++)
    {
        //尝试所有数字在此行上的可能性
        for(int k=0; k<16; k++)
        {
            int y = row(x,k);
            if(y == -2)return 0;//立即回溯
            if(y != -1)put_in(x,y,k);
        }
    }
    //遍历所有列
    for(int y=1; y<=16; y++)
    {
        //尝试所有数字在此列上的可能性
        for(int k=0; k<16; k++)
        {
            int x = col(y,k);
            if(x == -2)return 0;//立即回溯
            if(x != -1)put_in(x,y,k);
        }
    }
    //遍历所有4X4矩阵块
    for(int r=1; r<=16; r+=4)
    {
        for(int c=1; c<=16; c+=4)
        {
            for(int k=0; k<16; k++)
            {
                int x,y;//记录对应坐标
                cuble(r,c,k,x,y);
                if(x == -2)return 0;//立即回溯
                if(x != -1)put_in(r+x,c+y,k);
            }
        }
    }
    if(num==256)//如果所有点都已填上字符
    {
        print();
        return true;
    }
    //存储图，便于回溯
    int t_num;
    int t_maps[17][17];
    int t_table[17][17];
    t_num = num;
    for(int i=1; i<=16; i++)
    {
        for(int j=1; j<=16; j++)
        {
            t_maps[i][j] = maps[i][j];
            t_table[i][j] = table[i][j];
        }
    }
    //遍历未确定的点，找可填入的字母最少的位置
    int mx,my,mn=16;
    for(int i=1; i<=16; i++)
    {
        for(int j=1; j<=16; j++)
        {
            if(maps[i][j]>=0)continue;
            int r = 16 - cal(table[i][j]);
            if(r  < mn)
            {
                mn = r;
                mx = i,my = j;
            }
        }
    }
    //选择该点开始dfs，遍历所有可能的数字
    for(int k=0; k<16; k++)
    {
        if((table[mx][my] & 1<<k) == 0)
        {
            put_in(mx,my,k);
            if(dfs())return 1;//如果发现有解
            //回溯
            num = t_num;
            for(int i=1; i<=16; i++)
            {
                for(int j=1; j<=16; j++)
                {
                    maps[i][j] = t_maps[i][j];
                    table[i][j] = t_table[i][j];
                }
            }
        }
    }
    return 0;//回溯
}

int main()
{
    char s[20];
    while(~scanf("%s",s))
    {
        //初始化
        num=0;
        memset(table,0,sizeof table);
        //读图
        for(int j=1;j<=16;j++)
        {
            if(s[j-1]!='-')put_in(1,j,s[j-1]-'A');
            else maps[1][j]=-1;
        }
        for(int i=2; i<=16; i++)
        {
            scanf("%s",s);
            for(int j=1; j<=16; j++)
            {
                if(s[j-1]!='-')
                    put_in(i,j,s[j-1]-'A');
                else
                    maps[i][j] = -1;
            }
        }
        dfs();
    }
    return 0;
}

#include<iostream>
#include<unordered_map>
#include<algorithm>
#include<cstring>
#include<queue>
#include<cmath>
#include<set>

#define INF 0x3f3f3f3f
#define x first
#define y second

using namespace std;

typedef pair<int,int> PII;
typedef unsigned long long ull;
typedef long long ll;
const int N=16,M=1<<N;
int row[N],col[N],cell[4][4];//分别是：行中有没有使用，列中有没有使用，九宫格中有没有使用
char str[20][20];//读入每一行数据
int ones[M];//存储于每一个状态中有多少的数字
int map[M];//打表，表示2^n所代表的数值

void init()
{
     for(int i=0;i<N;i++)row[i]=col[i]=(i<<N)-1;//行和列中初始值全部设为1
     for (int i = 0; i < 4; i ++ )
     for (int j = 0; j < 4; j ++ )
     cell[i][j]=(1<<N)-1;//小格子中每位都是1
}

void draw(int x,int y,int t,bool flag)//在x，y中填上t，或是删掉
{
    if(flag)str[x][y]='A'+t;
    else str[x][y]='-';
    int v=1<<t;
    if(!flag)v=-v;
    row[x]-=v;
    col[y]-=v;
    cell[x/4][y/4]-=v;
}

int lowbit(int x)//返回末尾的1
{
    return x&-x;
}

int get(int x,int y)
{
    return row[x]&col[y]&cell[x/4][y/4];
}

bool dfs(int cnt)
{
    if(!cnt)return true;//若是当前没有-，说明找到了合法方案
    int min=N+1;//若是当前没有空格，说明找到了合法方案
    int x,y;//记录对应的格子
    for(int i=0;i<N;i++)
    for(int j=0;j<N;j++)
    if(str[i][j]=='-')
    {
        int state=get(i,j);
        if(ones[state]<min)
        {
            min=ones[state];
            x=i,y=j;
        }
    }
    int state=get(x,y);//找到该格子能填的数；
    for(int i=state;i;i-=lowbit(i))//枚举所有数字
    {
        int t=map[lowbit(i)];//从map中找到对应的数字
        draw(x,y,t,true);
        if(dfs(cnt-1))return true;
        draw(x,y,t,false);//回溯
    }
    return false;
}

int main()
{
    //打表
    for(int i=0;i<N;i++)map[1<<i]=i;
    for(int i=0;i<1<<N;i++)
    for(int j=0;j<N;j++)
    ones[i]+=i>>j&1;
    int cnt=0;//计算共有多少空位
    for(int i=0;i<16;i++)
    {
        scanf("%s",str[i]);
        for(int j=0;j<N;j++)
        if(str[i][j]=='-')cnt++;
        else
        {
            int t=str[i][j]-'A';
            draw(i,j,t,true);//存图
        }
    }
    dfs(cnt);
    for(int i=0;i<N;i++)
    puts(str[i]);
    return 0;
}

B - Wow! Such Conquering!

题目：

hdu-4848

思路：

难点：

理解题目，要求的是所浪费的时间和，即每次到达一个点时花费时间乘剩余点数之和

首先通过Floyd算法，计算两点之间的最短距离，然后dfs搜索答案。

剪枝：

1.若是已经花费的时间比最大的规定时间（deadline）还大，则即使该点是最后一个点，也无法满足题目要求

2.若是已经花费的时间比该点的规定时间（deadline）还大，无法满足题目要求，剪枝

3.若是已经浪费的时间比已知答案还要大，一定不是最优解，剪枝

4.如果存在点没有走过，且已走过时间+去该点所需要时间无法到达目的点，剪枝

5.不走已经走过的点，剪枝

代码：

#include<iostream>
#include<cstring>

using namespace std;
const int inf=0x3f3f3f3f;
const int N=35;
int n;//记录结点个数
int map[N][N];//存图
int dl[N];//记录每一个点的deadline时间
bool st[N];//记录该点是否走过
int maxdl;//记录最大deadline值
int ans;//记录答案

void floyd()//Floyd模板
{
    for(int k=1;k<=n;k++)
    for(int i=1;i<=n;i++)
    for(int j=1;j<=n;j++)
    {
        if(map[i][j]>map[i][k]+map[k][j])
        map[i][j]=map[i][k]+map[k][j];
    }
}

void dfs(int pos,int hasnum,int tim,int kk,int sumtim)//传入参数：位置，已走过的点数，已消耗的时间，未走过的点数，浪费的时间
{
    if(tim>maxdl)return;//1.已消耗的时间比deadline-max还要大，剪枝
    if(tim>dl[pos])return;//2.已消耗的时间大于该点的deadline，剪枝
    if(sumtim>ans)return;//3.如果已经走过的总时间大于已知答案的时间，那么一定比已知答案大，剪枝
    for(int i=2;i<=n;i++)//遍历所有点
    {
        if(!st[i]&&tim+map[pos][i]>dl[i])return;//4.如果存在点没有走过，且已走过时间+去该点所需要时间无法到达目的点，剪枝
    }
    if(hasnum==n)//如果已经走过所有点
    {
        if(sumtim<ans)ans=sumtim;//更新答案
        return ;
    }
    for(int i=2;i<=n;i++)
    {
        if(st[i])continue;//5.如果该点走过，则不考虑，剪枝
        st[i]=true;
        dfs(i,hasnum+1,tim+map[pos][i],kk-1,sumtim+map[pos][i]*kk);
        st[i]=false;//回溯
    }
}

int main()
{
    while(~scanf("%d",&n))
    {
        for(int i=1;i<=n;i++)//读入图
        {
            for(int j=1;j<=n;j++)
            {
                scanf("%d",&map[i][j]);
            }
        }
        maxdl=-1;//初始化
        for(int i=2;i<=n;i++)//查找所有deadline中的最大值
        {
            scanf("%d",&dl[i]);
            if(maxdl<dl[i])maxdl=dl[i];
        }
        floyd();//遍历多源汇最短路
        //初始化
        ans=inf;
        memset(st,0,sizeof st);//初始化st数组
        dfs(1,1,0,n-1,0);//开始搜索答案
        if(ans==inf)puts("-1");
        else printf("%d\n",ans);
    }
}

C - Sticks

题目：

HDU - 1455

Acwing-167

思路：

代码：

#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
const int N=70;
int n;
int w[N],sum,length;
bool st[N];//判断此小木棒有没有被使用过

bool dfs(int u,int s,int start)//传入：已有的木棍数量，正在组合的木棍长度，枚举到哪一个小木棍
{
    if(u*length==sum)return true;//如果总的数量*长度==总长度，则表示有解
    if(s==length)return dfs(u+1,0,0);//表示组合好一根木棍，该组合下一组了
    //剪枝3-1，i从start开始枚举
    for(int i=start;i<n;i++)
    {
        if(st[i])continue;//如果该点在之前组合木棍时被使用过了
        if(s+w[i]>length)continue;//可行性剪枝，如果超范围，则continue
        st[i]=true;
        if(dfs(u,s+w[i],i+1))return true;//搜索
        st[i]=false;//回溯

        //剪枝3-3，如果该木棍组合在第一个都组合不出来，直接return
        if(!s)return false;

        //剪枝3-4,如果是最后一根木棍，则直接return 
        if(s+w[i]==length)return false;

        //剪枝3-2，所有长度相等的木棍全都排除
        int j=i;
        while(j<n&&w[j]==w[i])j++;
        i=j-1;
    }
    return false;
}

int main()
{
     while(cin>>n,n)
     {
         sum=0;
         memset(st,0,sizeof st);
         for(int i=0;i<n;i++)
         {
             cin>>w[i];
             sum+=w[i];
         }

        //剪枝2.优化搜索顺序，从大到小枚举
        sort(w,w+n);
        reverse(w,w+n);

        length=1;//因为要求可能最小长度，所以从1开始
        while (true)
        {
            if(sum%length==0&&dfs(0,0,0))//剪枝1.必须能够被总长度整除
            {
                cout<<length<<endl;
                break;
            }
            length++;
        }
     }
    return 0;
}

D - Rise in Price

题目:

hdu-6981

思路：

Dfs题目，个人的剪枝是：如果该点已存在宝石数，宝石价格倍数严格大的记录，那么剪枝；若新的宝石数和宝石价格都比已记录的大，则更新该点记录的宝石数与价格。

大佬思路（dp+dfs）：

大佬代码（dp+dfs）：

#include<iostream>

using namespace std;
const int N=105;
int n;
int a[N][N],b[N][N],c[N][N];
int ans;

int get(int Sa,int Sb,int Cmax)
{
    return (Sa+Sb+Cmax)>>1;
}

void dfs(int x,int y,int Sa,int Sb)//传入横纵坐标，和已经到手的宝石数与倍数
{
    if(x==n&&y==n)
    {
        ans=max(ans,Sa*Sb);
        return;
    }
    int t=get(Sa,Sb,c[x+1][y]);//如果最终答案经过(x+1,y)
    if(x<n&&t*t>ans)//1.x如果到达边界，剪枝  2.如果答案小于等于已知解，剪枝
    {
        dfs(x+1,y,Sa+a[x+1][y],Sb+b[x+1][y]);
    }
    int t=get(Sa,Sb,c[x][y+1]);//如果最终答案经过(x,y+1)
    if(y<n&&t*t>ans)//1.y如果到达边界，剪枝  2.如果答案小于等于已知解，剪枝
    {
        dfs(x+1,y,Sa+a[x][y+1],Sb+b[x][y+1]);
    }
}

int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        //读入
        scanf("%d",&n);
        for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)
        scanf("%d",&a[i][j]);
        for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)
        scanf("%d",&b[i][j]);
        //dp,求每个点最大的A+B
        for(int i=n;i>=1;i--)
        for(int j=n;j>=1;j--)
        {
            c[i][j]=a[i][j]+b[i][j];
            if(i<n)c[i][j]=max(c[i][j],c[i+1][j]+a[i][j]+b[i][j]);
            if(j<n)c[i][j]=max(c[i][j],c[i][j+1]+a[i][j]+b[i][j]);
        }
        ans=0;
        //开搜
        dfs(1,1,a[1][1],b[1][1]);
        printf("%d\n",ans);
    }
}

E - Jerboas

题目：

HDU - 2437

思路：

只要总路程是k的倍数里面取个最小的，并且下标最小

设置dist(x,y)表示目的地为x，且已走过的路程%k为y时，路程的大小为dist(x,y)

代码：

#include<iostream>
#include<cstring>

using namespace std;
const int N=1e3+5,M=2e4+5;

int ans,bestend;
int h[N],ne[M],w[M],e[M],idx;//邻接表
bool st[N];
char str[N];
int a[N];
int n,m,s,k;
int dist[N][N];
void Init()//初始化
{
    idx=0;
    memset(h,-1,sizeof h);
    memset(st,0,sizeof st);
    memset(dist,-1,sizeof dist);
    ans=0x3f3f3f3f;
    bestend=0x3f3f3f3f;
}

void add(int a,int b,int c)//加边
{
    e[idx]=b,w[idx]=c,ne[idx]=h[a],h[a]=idx++;
}

void dfs(int sum,int u)
{
    //如果满足题目要求，更新
    if(sum%k==0&&a[u]==1&&(sum<ans||(sum==ans&&u<bestend)))
    {
        ans=sum;
        bestend=u;
        return ;
    }
    for(int i=h[u];~i;i=ne[i])
    {
        int j=e[i];
        int v=sum+w[i];
        if(dist[j][v%k]==-1||dist[j][v%k]>v)//如果该点没有走过or
        {
            dist[j][v%k]=v;
            dfs(v,j);
        }
    }
}

int main()
{
    int t;
    scanf("%d",&t);
    for(int i=1;i<=t;i++)
    {
        //初始化
        Init();
        //读入数据
        scanf("%d%d%d%d",&n,&m,&s,&k);
        scanf("%s",str);
        for(int i=0;i<strlen(str);i++)
        {
            if(str[i]=='T')a[i+1]=0;
            else a[i+1]=1;
        }
        for(int i=0;i<m;i++)
        {
            int a,b,c;
            scanf("%d%d%d",&a,&b,&c);
            add(a,b,c);
        }
        dfs(0,s);
        printf("Case %d: ",i);
        if(ans!=0x3f3f3f3f)
        {
            printf("%d %d\n",ans,bestend);
        }
        else
        {
            printf("-1 -1\n");
        }
    }
    return 0;
}

F - Herbs Gathering

题目：

HDU - 5887

思路：

这题看上去是0-1背包，但是时间t太大，0-1背包不好开这么大的数组，所以用搜索写，这里开2个后缀数组，分别是到当前位置i时i-n的时间总和与到当前位置i时i-n的价值总和，用于剪枝，即当前搜索到的位置所用时间+后缀时间>T即剪枝以及当前搜索到的位置所得到价值+后缀价值<=sum即剪枝，另外时间按从大到小排序，从小到大排序会超时..

代码：

#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;
typedef long long ll;
const int N=105;

int n,cnt,T;
ll sum;
ll rv[N];//当前位置i时i-n的时间总和
ll rt[N];//当前位置i时i-n的价值总和
struct node
{
    int t,v;
    bool operator<(const node &nd)const
    {
        return t>nd.t;//从大到小排序
    }
}nd[N];

void dfs(int i,ll uv,ll ut)//记录当前
{
    sum=max(uv,sum);//更新最大价值
    if(i==n)return ;//如果取到最后，返回
    if(sum>=uv+rv[i])return ;//1.当前搜索到的位置所得到价值+后缀价值<=sum,剪枝
    if(T>=ut+rt[i])//2.如果时间充裕到可取所有，那就全都要，剪枝
    {
        sum=max(uv+rv[i],sum);
        return ;
    }
    if(ut+nd[i].t<=T)//如果可以取，那就取
    {
        dfs(i+1,uv+nd[i].v,ut+nd[i].t);
    }
    dfs(i+1,uv,ut);//同时考虑不取的情况
}


int main()
{
    while(~scanf("%d%d",&n,&T))
    {
        cnt=0;
        sum=0;
        for(int i=0;i<n;i++)
        {
            scanf("%d%d",&nd[cnt].t,&nd[cnt].v);
            if(nd[cnt].t<=T)//如果该药草花费时间小于规定时间，表示可使用
            {
                cnt++;
            }
        }
        n=cnt;//更新实际药草数
        sort(nd,nd+n);//排序
        rv[n]=rt[n]=0;
        //记录后缀和
        for(int i=n-1;i>=0;i--)
        {
            rv[i]=rv[i+1]+nd[i].v;
            rt[i]=rt[i+1]+nd[i].t;
        }  
        dfs(0,0,0);
        printf("%lld\n",sum);
    }
    return 0;
}
